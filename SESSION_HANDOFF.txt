常用查詢
--------
確認 group members:

select
  tg.id as group_id,
  tg.name as group_name,
  gm.user_id,
  gm.role
from public.travel_groups tg
join public.group_members gm
  on gm.group_id = tg.id
order by tg.created_at desc;

確認 trips:

select
  id,
  group_id,
  city,
  country,
  flag,
  start_date,
  end_date,
  status,
  tags,
  legacy_key
from public.trips
order by created_at desc;

確認 RLS:

select
  tablename,
  rowsecurity
from pg_tables
where schemaname = 'public'
  and tablename in (
    'travel_groups',
    'group_members',
    'trips',
    'itinerary_days',
    'itinerary_items',
    'expenses',
    'checklist_items'
  );
